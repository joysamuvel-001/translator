# ── PyTorch / audio (CPU wheels — pinned for NeMo + pyannote compat) ──

#   pip install torch==2.6.0 torchaudio==2.6.0 --index-url https://download.pytorch.org/whl/cpu

# NumPy MUST stay below 2.0 — pyannote.audio's Inference class references

# which was removed in torchaudio 2.1+. 3.3.2 is compatible with the
# torch/torchaudio 2.6.0 pin above.
pyannote.audio==3.3.2
